        function selectAnswer(selectedIndex, isTimeUp = false) {

            stopTimer();

            const allOptionButtons = optionsContainer.querySelectorAll('button');

            surpriseCardBtn.disabled = true;

            

            const currentTeam = activeTeams[currentTeamIndex];


            if (isTimeUp) {

                showFeedback("Tempo esgotado!", 'red');

                wrongSound.triggerAttackRelease("C2", "0.4s");

                if (isSuddenDeath) currentTeam.eliminatedInSuddenDeath = true;

            } else if (selectedIndex >= 0) {

                if (selectedIndex === question.answer) {

                    correctSound.triggerAttackRelease("C4", "0.5s");

                    currentTeam.score += question.points;

                    showFeedback(`Correto! +${question.points} pontos!`, 'green');

                    allOptionButtons[selectedIndex].classList.add('correct');

                    const originalIndex = teams.findIndex(t => t.name === currentTeam.name);

                    document.getElementById(`score-${originalIndex}`).classList.add('score-update');

                } else {

                    wrongSound.triggerAttackRelease("C2", "0.4s");

                    if (isSuddenDeath) {

                        currentTeam.eliminatedInSuddenDeath = true;

                    }

                    showFeedback(`Errado!`, 'red');

                    allOptionButtons[selectedIndex].classList.add('incorrect');

                    allOptionButtons[question.answer].classList.add('correct');

                }

            }


            updateScoreboard();

            

            setTimeout(() => {

                allOptionButtons.forEach(btn => btn.classList.remove('correct', 'incorrect'));

                const originalIndex = teams.findIndex(t => t.name === currentTeam.name);

                document.getElementById(`score-${originalIndex}`)?.classList.remove('score-update');

                

                if (isSuddenDeath && currentTeamIndex === activeTeams.length - 1) {

                    const remainingInSuddenDeath = activeTeams.filter(team => !team.eliminatedInSuddenDeath);


                    if (remainingInSuddenDeath.length <= 1) {

                        activeTeams = remainingInSuddenDeath;

                        showFinalScreen();

                        return;

                        currentTeamIndex = -1;

                nextTurn();

                displayQuestion();

            }, 2500);

        function nextTurn() {

            }

            currentTeamIndex = (currentTeamIndex + 1) % activeTeams.length;

        }

        function checkForTiebreaker() {

            const sortedTeams = [...activeTeams].sort((a, b) => b.score - a.score);

            if (sortedTeams.length <= 1) {

                 activeTeams = sortedTeams;

                 showFinalScreen();

                 return;

            }

            const topScore = sortedTeams[0].score;

            const tiedTeams = sortedTeams.filter(team => team.score === topScore);


            if (tiedTeams.length > 1) {

                isSuddenDeath = true;

                activeTeams = tiedTeams; 

                activeTeams.forEach(t => t.eliminatedInSuddenDeath = false);

                currentTeamIndex = 0;

                showFeedback("EMPATE! Iniciando Morte Súbita!", 'yellow', 4000);

                setTimeout(displayQuestion, 4000);

            } else {

                activeTeams = [sortedTeams[0]];

                showFinalScreen();

            }

        }

                    modalDescription.innerHTML = `Sua equipe chegou à última rodada sem usar a surpresa. Uma carta foi puxada automaticamente: <strong class="block text-2xl mt-2">${randomCard.name}</strong>.`;

                    modalOptions.innerHTML = ''; 

                    const okBtn = document.createElement('button');

                    okBtn.textContent = 'OK';

                    okBtn.classList.add('btn', 'bg-[#8d6e63]', 'hover:bg-[#6d4c41]', 'w-full', 'p-3', 'text-white');

                    okBtn.onclick = () => executeCardAction(randomCard);

                    modalOptions.appendChild(okBtn);

                } else {

                     executeCardAction(randomCard);

                }

            }, 1500);

        }

        function executeCardAction(card) {

            activeTeams[currentTeamIndex].hasUsedSurprise = true;

            surpriseCardBtn.disabled = true;

            modalTitle.className = "text-4xl md:text-5xl font-bold text-[#6d4c41] mb-4";

            modalTitle.textContent = card.name;

            modalDescription.innerHTML = card.description;

            modalOptions.innerHTML = '';

            modalClose.style.display = 'block';

            


            const currentTeam = activeTeams[currentTeamIndex];

            let resultText = "";


            switch(card.action) {

                case "steal_points":

                case "give_points":

                    modalClose.style.display = 'none'; 

                    modalOptions.innerHTML = '';

                    teams.forEach((team) => {

                        if (team.name === currentTeam.name) return;


                        const teamBtn = document.createElement('button');

                        teamBtn.textContent = team.name;

                        teamBtn.classList.add('btn', 'bg-[#795548]', 'hover:bg-[#5d4037]', 'text-white', 'p-3');

                        teamBtn.onclick = () => {

                            if (card.action === "steal_points") {

                                const stolen = Math.min(card.value, team.score);

                                team.score -= stolen;

                                currentTeam.score += stolen;

                                resultText = `${currentTeam.name} coletou ${stolen} pontos de ${team.name}!`;

                            }

                            if (card.action === "give_points") {

                                const given = Math.min(card.value, currentTeam.score);

                                currentTeam.score -= given;

                                team.score += given;

                                resultText = `${currentTeam.name} ofertou ${given} pontos para ${team.name}!`;

                            }

                            updateScoreboard();

                            showCardResultInModal(resultText);

                        };

                        modalOptions.appendChild(teamBtn);

                    });

                    break;

                

                case "give_to_last":

                    const sortedGive = [...teams].filter(t => t.name !== currentTeam.name).sort((a,b) => a.score - b.score);

                    const lastPlaceTeamGive = sortedGive[0];

                    if (lastPlaceTeamGive) {

                        const given = Math.min(100, currentTeam.score);

                        currentTeam.score -= given;

                        lastPlaceTeamGive.score += given;

                        resultText = `${currentTeam.name} ajudou ${lastPlaceTeamGive.name} com ${given} pontos!`;

                    } else {

                         resultText = "Não há outras equipes para ajudar! Nada acontece.";

                    }

                    updateScoreboard();

                    showCardResultInModal(resultText);

                    break;


                case "swap_with_first":

                    const sortedSwap = [...teams].filter(t => t.name !== currentTeam.name).sort((a,b) => b.score - a.score);

                    const firstPlaceTeam = sortedSwap[0];

                    if (firstPlaceTeam && firstPlaceTeam.score > currentTeam.score) {

                        [currentTeam.score, firstPlaceTeam.score] = [firstPlaceTeam.score, currentTeam.score];

                         resultText = `${currentTeam.name} trocou de pontuação com o líder ${firstPlaceTeam.name}!`;

                    } else {

                        resultText = "Não há um líder com mais pontos para trocar! Nada acontece.";

                    }

                    updateScoreboard();

                    showCardResultInModal(resultText);

                    break;


                case "bonus_points":

                    currentTeam.score += card.value;

                    resultText = `Bênção recebida! +${card.value} pontos!`;

                    updateScoreboard();

                    showCardResultInModal(resultText);

                    break;

                    

                case "lose_points":

                    currentTeam.score = Math.max(0, currentTeam.score - card.value);

                    resultText = `Que provação! -${card.value} pontos!`;

                    updateScoreboard();

                    showCardResultInModal(resultText);

                    break;

            }

        }
